# OAuth Hub Environment Configuration for Oracle Cloud
# Copy this to .env and fill in your values

# Server Configuration
NODE_ENV=production
PORT=3001
HOST=0.0.0.0
HTTPS=true

# OAuth Configuration
OAUTH_ISSUER=https://your-oracle-server.com:3001
OAUTH_CLIENT_ID=oauth-hub
OAUTH_CLIENT_SECRET=your-generated-secret

# Database Configuration (Oracle Cloud)
DATABASE_URL=postgresql://username:password@oracle-db-host:5432/oauth_hub

# SSL Certificates (for HTTPS)
SSL_CERT_PATH=/etc/ssl/certs/oauth-hub.crt
SSL_KEY_PATH=/etc/ssl/private/oauth-hub.key

# Session Configuration
SESSION_SECRET=your-secure-random-session-secret

# Logging
LOG_LEVEL=info
LOG_FILE=/var/log/oauth-hub.log

# Rate Limiting
RATE_LIMIT_WINDOW=900000
RATE_LIMIT_MAX=100

# CORS
ALLOWED_ORIGINS=https://your-frontend.com,https://another-domain.com

# Oracle Cloud Specific
ORACLE_REGION=us-ashburn-1
ORACLE_COMPARTMENT_ID=your-compartment-id
